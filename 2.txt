// input: a[]
swap (int a, int b) {
    int tg = a;
    a = b;
    b = tg;
}
SapXep (int a[], int n) {
    for (int i = 0; i < n; i++){
        for (int j = i + 1; j < n; j++){
            if (a[i] < a[j]) {
                swap (a[i], a[j]);
            }
        }        
    }
}
